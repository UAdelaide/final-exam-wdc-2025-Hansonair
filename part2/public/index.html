const express = require('express'); const router = express.Router(); // Other routes... router.post('/logout', (req, res) => { req.session.destroy(err => { if (err) { return res.status(500).json({ error: 'Failed to logout' }); } res.clearCookie('connect.sid', { path: '/' }); res.json({ message: 'Logged out successfully' }); }); }); module.exports = router;
